CPU BENCHMARK (3 seconds, runs in browser):
async function runCPUBenchmark() {
  const duration = 3000;
  const start = performance.now();
  let ops = 0;
  while (performance.now() - start < duration) {
    for (let i = 0; i < 100000; i++) {
      Math.fround(i * 1.23456);
    }
    ops++;
  }
  const elapsed = (performance.now() - start) / 1000;
  return { ops_per_sec: ops / elapsed };
}

MEMORY BANDWIDTH TEST:
async function runMemoryBenchmark() {
  const buf = new Float32Array(32 * 1024 * 1024); // 128MB
  buf.fill(1.0);
  const start = performance.now();
  let sum = 0;
  for (let i = 0; i < buf.length; i++) sum += buf[i];
  const elapsed = (performance.now() - start) / 1000;
  return { 
    bandwidth_gb_s: (buf.byteLength / 1e9) / elapsed 
  };
}

LOCAL STORAGE SPEED TEST (IndexedDB, not CDN fetch):
// DO NOT use fetch() for disk test. That measures 
// internet speed, not local storage speed.
// Use IndexedDB write then read instead:
async function runStorageBenchmark() {
  const size = 20 * 1024 * 1024; // 20MB
  const data = new Uint8Array(size).fill(42);
  
  // Write to IndexedDB
  const writeStart = performance.now();
  await idbWrite('llmeter-bench', data);
  const writeTime = performance.now() - writeStart;
  
  // Read back from IndexedDB
  const readStart = performance.now();
  await idbRead('llmeter-bench');
  const readTime = performance.now() - readStart;
  
  return {
    write_mb_s: (size / 1e6) / (writeTime / 1000),
    read_mb_s:  (size / 1e6) / (readTime / 1000),
  };
}

COMBINE INTO real_perf_factor:
cpu_factor  = clamp(ops_per_sec / 50000, 0.5, 2.0)
mem_factor  = clamp(bandwidth_gb_s / 20.0, 0.4, 1.8)
disk_factor = clamp(read_mb_s / 500, 0.3, 1.5)
real_perf_factor = (cpu_factor * 0.5) + (mem_factor * 0.35) + (disk_factor * 0.15)

PHASE 6: TOKENS/SEC ESTIMATE

Separate CPU and GPU paths. Never combine into one fake number.

function estimateTPS(deviceProfile, contextSize, chunked) {
  const { cores, simd, threads, webgpu, unifiedMemory,
          bandwidth_gb_s, real_perf_factor } = deviceProfile;
  
  // CPU baseline from benchmark
  let cpu_tps = cores * 2.0 * real_perf_factor;
  if (simd)    cpu_tps *= 1.5;
  if (threads) cpu_tps *= 1.2;
  
  // Memory bandwidth multiplier
  const mem_mult = clamp(bandwidth_gb_s / 25.0, 0.5, 1.5);
  cpu_tps *= mem_mult;
  
  // GPU path
  let gpu_tps = 0;
  if (unifiedMemory) gpu_tps = cpu_tps * 3.0;
  else if (webgpu)   gpu_tps = cpu_tps * 2.5;
  
  let effective = Math.max(cpu_tps, gpu_tps);
  
  // Context penalty
  if (contextSize > 8192)      effective *= 0.55;
  else if (contextSize > 4096) effective *= 0.70;
  else if (contextSize > 2048) effective *= 0.85;
  
  // Chunked penalty
  if (chunked) effective *= 0.80;
  
  return {
    low:  Math.round(effective * 0.65),
    high: Math.round(effective * 1.20),
    path: gpu_tps > cpu_tps ? "GPU" : "CPU",
    limiting_factor: mem_mult < 0.8 ? "memory bandwidth" : "compute"
  };
}

PHASE 7: LOADING STRATEGY PLANNER

Name it "Loading Strategy Planner". Not "simulator".
It plans. It does not download real model weights.

function planLoadingStrategy(model_size_gb, usable_ram_gb, 
                              disk_read_mb_s) {
  const overhead = 0.35;
  const safe_ram = usable_ram_gb - overhead;
  const spike    = model_size_gb * 1.25;
  
  let strategy, chunks, chunk_size;
  
  if (spike <= safe_ram) {
    strategy   = "direct";
    chunks     = 1;
    chunk_size = model_size_gb;
  } else {
    chunk_size = safe_ram * 0.75;
    chunks     = Math.ceil(model_size_gb / chunk_size);
    strategy   = "chunked";
  }
  
  const load_time_sec = (model_size_gb / chunks) / 
                        (disk_read_mb_s / 1000) * chunks * 1.15;
  
  const failure_prob = Math.max(0,
    (spike - safe_ram) / (safe_ram + 0.1)
  );
  
  return {
    strategy,
    chunks,
    chunk_size_gb: +(model_size_gb / chunks).toFixed(2),
    est_load_time_sec: Math.round(load_time_sec),
    worst_case_ram_gb: +Math.min(spike, usable_ram_gb * 0.95).toFixed(1),
    failure_probability: +Math.min(failure_prob, 0.95).toFixed(2),
    risk: failure_prob > 0.6 ? "high" 
        : failure_prob > 0.35 ? "medium" : "low",
  };
}

function encodeReport(result) {
  const payload = {
    s:  result.final_score,
    r:  result.ram_gb,
    c:  result.cores,
    a:  result.arch,
    g:  result.gpu_renderer?.slice(0, 30),
    t:  result.tps_low + '-' + result.tps_high,
    f:  result.top_fits.slice(0,3).map(m => m.name),
    ts: Date.now(),
  };
  return btoa(JSON.stringify(payload));
}
